# ==============================================================================
# 4. TERRAFORM LIFE-CYCLE EXECUTION MANAGEMENT
# ==============================================================================

resource "terracurl_request" "trigger_workflow" {
  name   = "trigger_phase2_execution"
  url    = "https://workflowexecutions.googleapis.com/v1/projects/${var.gcp_project_id}/locations/${var.gcp_region}/workflows/${var.workflow_name}/executions"
  method = "POST"
  
  headers = {
    Authorization = "Bearer ${data.google_client_config.current.access_token}"
    Content-Type  = "application/json"
  }

  response_codes = [200, 201]

  request_body = jsonencode({
    argument = jsonencode(merge(
      local.workflow_template,
      {
        phase_name = var.phase_name
        env        = var.environment
        retries    = var.retries
      }
    ))
  })

  # --- SECURE DESTRUCTION BYPASS ---
  # We point the destroy hook safely to the general workflow execution listing endpoint using a GET request.
  # This acts as a completely non-destructive read validation that satisfies the local engine sandbox 
  # without needing problematic egress routing or dynamic state interpolation out to public targets.
  destroy_url    = "https://workflowexecutions.googleapis.com/v1/projects/${var.gcp_project_id}/locations/${var.gcp_region}/workflows/${var.workflow_name}/executions"
  destroy_method = "GET"
  
  destroy_headers = {
    Authorization = "Bearer ${data.google_client_config.current.access_token}"
  }

  destroy_response_codes = [200, 404]
  
  lifecycle {
    replace_triggered_by = [terraform_data.always_run]
  }
}


#Used to enforce execution replacement loops across apply actions
resource "terraform_data" "always_run" {
  input = timestamp()
}

# Assures asynchronous stability processing buffer before state calculations resume
resource "time_sleep" "wait_for_workflow" {
  depends_on      = [terracurl_request.trigger_workflow]
  create_duration = "20s" 
}
